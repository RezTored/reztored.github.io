// ============================================================
// REGLAS DE FIRESTORE PARA EL POKER — agregar dentro de tu bloque
// service cloud.firestore { match /databases/{database}/documents { ... } }
// junto a las reglas que ya tenías para "users", "usernames", "opinions", etc.
//
// Dónde pegarlas: Firebase Console → Firestore Database → Reglas.
// Si ya tenés reglas para otras colecciones, agregá SOLO el bloque
// "match /pokerRooms/{roomId} { ... }" de acá abajo dentro de tus
// reglas existentes (no borres lo que ya tenías).
// ============================================================

match /pokerRooms/{roomId} {

  // El documento principal de la sala (jugadores, pozo, fase, etc.)
  // es público de lectura (así todos ven la mesa) y cualquier usuario
  // logueado puede escribirlo. Es el mismo modelo de confianza que ya
  // usás para los likes y las donaciones de petoCoins: la app de por
  // sí valida las reglas del juego, esto es la puerta de entrada.
  allow read: if true;
  allow create: if request.auth != null;
  allow update: if request.auth != null;
  allow delete: if false;

  // El mazo restante (después de repartir). SOLO el creador de la
  // sala lo puede leer o escribir. Esto es lo que evita que un jugador
  // "espíe" las próximas cartas comunitarias abriendo la consola.
  match /dealer/{docId} {
    allow read, write: if request.auth != null &&
      request.auth.uid == get(/databases/$(database)/documents/pokerRooms/$(roomId)).data.hostUid;
  }

  // La mano (2 cartas) de cada jugador. Solo el dueño la puede leer.
  // Solo el creador de la sala la puede escribir (es quien reparte).
  match /hands/{uid} {
    allow read: if request.auth != null && request.auth.uid == uid;
    allow write: if request.auth != null &&
      request.auth.uid == get(/databases/$(database)/documents/pokerRooms/$(roomId)).data.hostUid;
  }

  // Cartas reveladas en el showdown. Lectura pública (para mostrar a
  // todos quién tenía qué). Solo el dueño puede publicar SU revelación,
  // y tiene que coincidir exactamente con lo que se le repartió (así
  // nadie puede "mentir" sobre sus propias cartas al mostrar).
  match /reveals/{uid} {
    allow read: if true;
    allow create: if request.auth != null && request.auth.uid == uid &&
      request.resource.data.cards == get(/databases/$(database)/documents/pokerRooms/$(roomId)/hands/$(uid)).data.cards;
    allow update: if false;
    allow delete: if request.auth != null &&
      request.auth.uid == get(/databases/$(database)/documents/pokerRooms/$(roomId)).data.hostUid;
  }
}
